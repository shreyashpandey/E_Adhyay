1    package com.example.anonymus.e_adhyay; 
2     
3    import android.animation.ObjectAnimator; 
4    import android.content.Intent; 
5    import android.graphics.Color; 
6    import android.os.Build; 
7    import android.os.Handler; 
8    import android.support.v7.app.AppCompatActivity; 
9    import android.os.Bundle; 
10   import android.view.View; 
11   import android.view.Window; 
12   import android.view.WindowManager; 
13   import android.widget.ProgressBar; 
14    
15   public class SplashActivity extends AppCompatActivity { 
16    
17       ProgressBar splashProgress; 
18       int SPLASH_TIME = 3000; //This is 3 seconds 
19    
20       @Override 
21       protected void onCreate(Bundle savedInstanceState) { 
22           super.onCreate(savedInstanceState); 
23           if (Build.VERSION.SDK_INT >= 21) { 
24               getWindow().getDecorView().setSystemUiVisibility(View.SYSTEM_UI_FLAG_LAYOUT_STABLE | View.SYSTEM_UI_FLAG_LAYOUT_FULLSCREEN); 
25           } 
26           if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) { 
27               Window window = getWindow(); 
28               window.addFlags(WindowManager.LayoutParams.FLAG_DRAWS_SYSTEM_BAR_BACKGROUNDS); 
29               window.setStatusBarColor(Color.TRANSPARENT); 
30           } 
31           setContentView(R.layout.activity_splash); 
32    
33           //This is additional feature, used to run a progress bar 
34           splashProgress = findViewById(R.id.splashProgress); 
35           playProgress(); 
36    
37           //Code to start timer and take action after the timer ends 
38           new Handler().postDelayed(new Runnable() { 
39               @Override 
40               public void run() { 
41                   //Do any action here. Now we are moving to next page 
42                   Intent mySuperIntent = new Intent(SplashActivity.this, HomeActivity.class); 
43                   startActivity(mySuperIntent); 
44    
45                   //This 'finish()' is for exiting the app when back button pressed from Home page which is ActivityHome 
46                   finish(); 
47    
48               } 
49           }, SPLASH_TIME); 
50       } 
51    
52       //Method to run progress bar for 5 seconds 
53       private void playProgress() { 
54           ObjectAnimator.ofInt(splashProgress, "progress", 100) 
55                   .setDuration(5000) 
56                   .start(); 
57       } 
58   }
